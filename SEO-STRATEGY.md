### 2.2.3 Duplicate/Thin Content Prevention

```typescript
// KEY RULES for preventing thin content penalties:
// 1. dataQualityScore < 50 → noindex, canonical to parent page
// 2. Every page must have MINIMUM 400 unique words beyond template
// 3. Testimonials MUST be unique per page (or omitted)
// 4. City stats MUST come from real Supabase data
// 5. Industry data MUST reference real merchant counts
// 6. Internal links must be contextually relevant (not random)

/**
 * Content quality gatekeeper - enforces Google Helpful Content standards
 */
export function evaluateProgrammaticPageQuality(
  pageType: 'city' | 'industry' | 'country' | 'variant',
  data: Record<string, unknown>
): { indexable: boolean; canonical: string; reason: string } {
  const score = calculatePageQuality(pageType, data);
  const thresholds: Record<string, number> = {
    city: 50, industry: 40, country: 60, variant: 45,
  };
  
  const threshold = thresholds[pageType];
  
  if (score < threshold) {
    const parentUrls: Record<string, string> = {
      city: '/whatsapp-store',
      industry: '/ecommerce',
      country: '/whatsapp-store',
      variant: '/free-store-builder',
    };
    return {
      indexable: false,
      canonical: parentUrls[pageType],
      reason: `Data quality score ${score} below threshold ${threshold}. Canonical to parent.`,
    };
  }
  
  return {
    indexable: true,
    canonical: '', // self-referencing
    reason: `Data quality score ${score} meets threshold ${threshold}. Indexable.`,
  };
}
```

```typescript
/**
 * FILE: lib/seo/indexnow.ts (NEW)
 * 
 * IndexNow protocol for instant indexing.
 * Submits URLs to Bing, Yandex, Seznam, and Naver simultaneously.
 * 
 * @see https://www.indexnow.org/
 * 
 * Usage: Call when a page is published, updated, or deleted.
 */

const INDEXNOW_KEY = process.env.INDEXNOW_API_KEY || "flowauxi2024seo";
const INDEXNOW_HOST = "www.flowauxi.com";

const SEARCH_ENGINES = [
  "https://api.indexnow.org/indexnow",      // Microsoft Bing
  "https://searchengines.yandex.ru/indexnow", // Yandex
  "https://search.seznam.cz/indexnow",       // Seznam
  "https://api.naver.com/indexnow",           // Naver
];

interface IndexNowPayload {
  host: string;
  key: string;
  keyLocation: string;
  urlList: string[];
}

export async function submitToIndexNow(urls: string[]): Promise<boolean[]> {
  const payload: IndexNowPayload = {
    host: INDEXNOW_HOST,
    key: INDEXNOW_KEY,
    keyLocation: `https://${INDEXNOW_HOST}/${INDEXNOW_KEY}.txt`,
    urlList: urls.slice(0, 10_000), // Max 10,000 URLs per request
  };

  const results = await Promise.allSettled(
    SEARCH_ENGINES.map(async (endpoint) => {
      const response = await fetch(endpoint, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload),
      });
      return response.ok;
    })
  );

  return results.map(r => r.status === "fulfilled" && r.value === true);
}

/**
 * Submit a single URL when content changes.
 * Call this from API routes when:
 * - Blog post published/updated
 * - Store page updated
 * - New programmatic page generated
 * - Landing page published
 */
export async function notifySearchEngines(url: string): Promise<void> {
  // Also submit to Google via Indexing API (requires OAuth)
  await submitToIndexNow([url]);
}

/**
 * Batch submit when new pages are generated.
 * Use for programmatic SEO batch operations.
 */
export async function batchNotifySearchEngines(urls: string[]): Promise<void> {
  // Submit in batches of 100
  for (let i = 0; i < urls.length; i += 100) {
    const batch = urls.slice(i, i + 100);
    await submitToIndexNow(batch);
    // Rate limit: wait 1 second between batches
    await new Promise(resolve => setTimeout(resolve, 1000));
  }
}
```
